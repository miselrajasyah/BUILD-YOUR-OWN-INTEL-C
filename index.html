// ========================================
// INTEL-C: Smart Home IoT System
// Intelligent Environmental Control
// ========================================

// Include Libraries
#include <DHT.h>
#include <Servo.h>
#include <LiquidCrystal_I2C.h>

// Pin Definitions
#define DHT_PIN_1 2     // DHT22 Lantai 1
#define DHT_PIN_2 3     // DHT22 Lantai 2
#define RAIN_PIN 4       // Rain Sensor
#define PIR_PIN 5        // PIR Motion Sensor
#define FAN_PIN 6        // Exhaust Fan
#define SERVO_WINDOW 9  // Servo Jendela
#define SERVO_CANOPY 10 // Servo Kanopi

// Sensor & Actuator Objects
DHT dht1(DHT_PIN_1, DHT22);
DHT dht2(DHT_PIN_2, DHT22);
Servo servoWindow;
Servo servoCanopy;
LiquidCrystal_I2C lcd(0x27, 16, 2);

// Threshold Values
const float TEMP_HIGH = 32.0;   // Suhu tinggi (¬∞C)
const float TEMP_LOW = 24.0;    // Suhu rendah (¬∞C)
const float HUMIDITY_HIGH = 75.0; // Kelembaban tinggi (%)

void setup() {
  // Initialize Serial Monitor
  Serial.begin(9600);
  
  // Initialize Sensors
  dht1.begin();
  dht2.begin();
  
  // Initialize LCD
  lcd.init();
  lcd.backlight();
  lcd.setCursor(0, 0);
  lcd.print("INTEL-C System");
  lcd.setCursor(0, 1);
  lcd.print("Initializing...");
  delay(2000);
  
  // Initialize Pins
  pinMode(RAIN_PIN, INPUT);
  pinMode(PIR_PIN, INPUT);
  pinMode(FAN_PIN, OUTPUT);
  
  // Initialize Servos
  servoWindow.attach(SERVO_WINDOW);
  servoCanopy.attach(SERVO_CANOPY);
  
  // Set initial positions
  servoWindow.write(45);  // Semi-open
  servoCanopy.write(90);  // Open
  
  digitalWrite(FAN_PIN, LOW);
  
  lcd.clear();
}

void loop() {
  // Read Sensors
  float temp1 = dht1.readTemperature();
  float humidity1 = dht1.readHumidity();
  float temp2 = dht2.readTemperature();
  float humidity2 = dht2.readHumidity();
  
  // Average values
  float avgTemp = (temp1 + temp2) / 2.0;
  float avgHumidity = (humidity1 + humidity2) / 2.0;
  
  // Read digital sensors
  bool isRaining = !digitalRead(RAIN_PIN);  // Inverted logic
  bool motionDetected = digitalRead(PIR_PIN);
  
  // Decision Making Logic
  if (isRaining) {
    // Rain Protection Mode
    servoWindow.write(0);    // Close windows
    servoCanopy.write(0);    // Close canopy
    digitalWrite(FAN_PIN, LOW);
    
    lcd.setCursor(0, 1);
    lcd.print("Mode: RAIN     ");
  } 
  else if (avgTemp > TEMP_HIGH) {
    // Cooling Mode
    servoWindow.write(90);   // Open windows
    servoCanopy.write(90);   // Open canopy
    digitalWrite(FAN_PIN, HIGH);
    
    lcd.setCursor(0, 1);
    lcd.print("Mode: COOLING  ");
  } 
  else if (avgTemp < TEMP_LOW) {
    // Energy Saving Mode
    servoWindow.write(0);    // Close windows
    servoCanopy.write(45);   // Semi-close
    digitalWrite(FAN_PIN, LOW);
    
    lcd.setCursor(0, 1);
    lcd.print("Mode: SAVING   ");
  }
  else {
    // Normal Mode
    servoWindow.write(45);   // Semi-open
    servoCanopy.write(90);   // Open
    
    if (avgHumidity > HUMIDITY_HIGH) {
      digitalWrite(FAN_PIN, HIGH);
    } else {
      digitalWrite(FAN_PIN, LOW);
    }
    
    lcd.setCursor(0, 1);
    lcd.print("Mode: NORMAL   ");
  }
  
  // Display on LCD
  lcd.setCursor(0, 0);
  lcd.print("T:");
  lcd.print(avgTemp, 1);
  lcd.print("C H:");
  lcd.print(avgHumidity, 0);
  lcd.print("%  ");
  
  // Debug to Serial Monitor
  Serial.print("Temp: ");
  Serial.print(avgTemp);
  Serial.print(" | Humidity: ");
  Serial.print(avgHumidity);
  Serial.print(" | Rain: ");
  Serial.println(isRaining ? "YES" : "NO");
  
  delay(2000);  // Update every 2 seconds
}

üìñ Penjelasan Kode:

1. Libraries: DHT.h untuk sensor suhu, Servo.h untuk motor, LiquidCrystal_I2C.h untuk LCD
2. Pin Definitions: Tentukan pin Arduino untuk setiap komponen
3. Setup(): Inisialisasi sensor, servo, dan LCD saat Arduino menyala
4. Loop(): Baca sensor ‚Üí Buat keputusan ‚Üí Kontrol aktuator ‚Üí Tampilkan data
5. Logika IF-ELSE: Jika hujan tutup semua, jika panas buka & nyalakan fan, jika dingin tutup semua

üõ†Ô∏è Tutorial Step-by-Step

Ikuti langkah-langkah ini untuk membuat INTEL-C kamu!

1

Persiapan & Perhitungan

Mulai dengan menghitung dimensi rumah menggunakan skala 1:50. Buat sketsa desain di kertas dan tentukan posisi jendela, pintu, dan sensor.

üí° Tips: Gunakan kertas grafik untuk sketsa lebih presisi. Pastikan semua anggota kelompok setuju dengan desain sebelum mulai konstruksi.
2

Konstruksi Struktur Rumah

Potong impraboard sesuai dimensi yang sudah dihitung. Rakit dinding, lantai, dan atap menggunakan lem tembak. Buat bukaan untuk jendela dan pintu.

üí° Tips: Potong dengan hati-hati menggunakan cutter tajam. Gunakan penggaris untuk hasil potongan yang rapi dan presisi.
3

Instalasi Servo Motor

Pasang servo motor untuk jendela dan kanopi. Buat mekanisme penggerak menggunakan kawat atau tali nilon. Test gerakan servo 0-90 derajat.

üí° Tips: Test servo terlebih dahulu di breadboard sebelum memasang permanent. Pastikan gerakan smooth tanpa hambatan.
4

Wiring & Koneksi Elektronik

Hubungkan semua sensor dan aktuator ke Arduino sesuai diagram wiring. Gunakan breadboard untuk koneksi yang rapi. Cek setiap koneksi dengan multimeter.

üí° Tips: Gunakan kabel jumper dengan warna berbeda untuk setiap jenis koneksi (merah=5V, hitam=GND, warna lain=Signal).
5

Upload & Testing Kode

Upload kode Arduino ke board. Test setiap sensor satu per satu. Cek output di Serial Monitor. Pastikan semua komponen merespons dengan benar.

üí° Tips: Jika ada error, cek koneksi pin dan library yang terinstall. Gunakan Serial.print() untuk debugging.
6

Integrasi & Fine-tuning

Integrasikan semua komponen dalam rumah miniatur. Adjust threshold value di kode sesuai kondisi ruangan. Test dengan kondisi cuaca simulasi.

üí° Tips: Simulasikan hujan dengan spray air, panas dengan hairdryer. Catat respons sistem dan adjust jika perlu.
7

Finishing & Dokumentasi

Cat rumah dengan warna menarik. Pasang LCD display di posisi yang visible. Buat video demonstrasi cara kerja sistem. Siapkan presentasi!

üí° Tips: Dokumentasikan setiap tahap dengan foto dan video. Ini akan berguna untuk presentasi dan portfolio kalian!
