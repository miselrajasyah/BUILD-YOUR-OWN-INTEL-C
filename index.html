üíª Kode Arduino

// INTEL-C: Smart Home IoT System
#include <DHT.h>
#include <Servo.h>
#include <LiquidCrystal_I2C.h>

#define DHT_PIN_1 2
#define RAIN_PIN 4
#define PIR_PIN 5
#define FAN_PIN 6
#define SERVO_WINDOW 9

DHT dht1(DHT_PIN_1, DHT22);
Servo servoWindow;
LiquidCrystal_I2C lcd(0x27, 16, 2);

void setup() {
  dht1.begin();
  lcd.init();
  lcd.backlight();
  pinMode(RAIN_PIN, INPUT);
  servoWindow.attach(SERVO_WINDOW);
}

void loop() {
  float temp = dht1.readTemperature();
  bool isRaining = !digitalRead(RAIN_PIN);
  
  if (isRaining) {
    servoWindow.write(0); // Tutup
  } else if (temp > 32) {
    servoWindow.write(90); // Buka
  }
  delay(2000);
}
